from flask import Flask, render_template, request, jsonify import pickle import numpy as np app = Flask(__name__) # Load the model with open('random_forest_model.pkl', 'rb') as f: model = pickle.load(f) # Serve the form @app.route('/', methods=['GET']) def home(): return render_template('index.html') # Prediction route @app.route('/predict', methods=['POST']) def predict(): try: # Get form data data = request.form features = [float(data[f'feature{i}']) for i in range(1, 6)] # Adjust for your model's features input_array = np.array([features]) # Reshape into (1, n_features) # Make prediction prediction = model.predict(input_array)[0] # Render the result on the same page return render_template('index.html', prediction=prediction) except Exception as e: return jsonify({'error': str(e)}), 500 if __name__ == '__main__': app.run(debug=True)
